    Option name Type Description
    settings Object A settings object with a client_id member

    Jamendo API client class
    Use it with
    var client = new Jamendo({ client_id: XXXXXX })

    var Jamendo = function(settings) {
  if (!settings || !settings.client_id) {
    throw 'You must provide a client_id setting';
  }

  this.base_url = 'http://api.jamendo.com/' + (settings.version ? settings.version : 'v3.0');
  this.client_id = settings.client_id;
};

module.exports = Jamendo;

    clean_params

    method
    Jamendo.prototype.clean_params() -> Object

    Option name Type Description
    path String The path parameters will be used against

    Clean a object parameters according to Jamendo policy
    - client_id is set if not specified
    - limit and offset are set if not specified
    - arrays are converted to space-separated strings

    Jamendo.prototype.clean_params = function(path, object){
  var string_params = [ 'id', 'artist_id', 'order' ];
  var param, default_params = {
    limit     : 10,
    offest    : 0,
    format    : 'json',
    client_id : this.client_id, 
  };

  // arrays as strings
  for (var i = 0; i < string_params.length; i++) {
    param = string_params[i];
    if (util.isArray(object[param])) {
      object[param] = object[param].join(' ');
    }
  }

  // explicit params
  for (var name in default_params) {
    if (typeof object[name] === 'undefined') {
      object[name] = default_params[name];
    }
  }

  return object;
};

    Jamendo.prototype.request() -> Request

    Option name Type Description
    path String The path to the api endpoint

    Main request wrapper

    Jamendo.prototype.request = function(path, parameters, callback) {
  parameters = this.clean_params(path, parameters || {});

  return request({
    url: this.base_url + path,
    method: 'GET',
    qs: parameters,
    json: true
  }, function(error, response, body){
    callback(error, body);
  });
};
